kde4_header()

add_subdirectory( about ) 

include_directories( ${CMAKE_SOURCE_DIR}/kontact/interfaces ${KDE4_INCLUDE_DIR} ${QT_INCLUDES} ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR}  )


########### next target ###############

set(kontact_LIB_SRCS 
   main.cpp 
   mainwindow.cpp 
   sidepanebase.cpp 
   iconsidepane.cpp 
   aboutdialog.cpp )

kde4_automoc(${kontact_LIB_SRCS})

kde4_add_kcfg_files(kontact_LIB_SRCS prefs.kcfgc )

kde4_add_library(kontact SHARED ${kontact_LIB_SRCS})

target_link_libraries(kontact  ${QT_AND_KDECORE_LIBS} kdepim kpinterfaces kontact kparts kutils khtml )

set_target_properties(kontact PROPERTIES VERSION 1.0.0 SOVERSION 1 )
install_targets(${LIB_INSTALL_DIR} kontact )


########### next target ###############

set(kontact_SRCS dummy.cpp )

kde4_automoc(${kontact_SRCS})

file(WRITE dummy.cpp "//autogenerated file by cmake\n")
kde4_add_executable(kontact ${kontact_SRCS})

target_link_libraries(kontact  ${QT_AND_KDECORE_LIBS} )

install_targets(/bin kontact )


########### next target ###############

set(kcm_kontact_PART_SRCS kcmkontact.cpp )

kde4_automoc(${kcm_kontact_PART_SRCS})

kde4_add_plugin(kcm_kontact ${kcm_kontact_PART_SRCS})

kde4_install_libtool_file( ${PLUGIN_INSTALL_DIR} kcm_kontact )

target_link_libraries(kcm_kontact  ${QT_AND_KDECORE_LIBS} kontact kdepim )

install_targets(${LIB_INSTALL_DIR}/kde4 kcm_kontact )


########### install files ###############

install_files( ${XDG_APPS_DIR} FILES Kontact.desktop )
install_files( ${KCFG_INSTALL_DIR} FILES kontact.kcfg )
install_files( ${SERVICES_INSTALL_DIR} FILES kontactconfig.desktop )
install_files( ${DATA_INSTALL_DIR}/kontact FILES kontactui.rc )
install_files( ${DATA_INSTALL_DIR}/kontact FILES kontact.setdlg )

kde4_install_icons( ${ICON_INSTALL_DIR}  hicolor )

kde4_footer()
